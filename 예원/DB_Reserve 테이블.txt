use apidb;

CREATE TABLE Reserve(
	id varchar(50),				-- 아이디
    service varchar(50),		-- 서비스 종류(돌봄,산책,미용)
    time varchar(50),			-- 돌봄/산책 시간(60분,120분,180분)
    grooming varchar(50),		-- 미용(전체/위생)
    pet varchar(50),			-- 반려동물 종류(소/중형,대형,고양이)
    date varchar(50),			-- 예약날짜
    message varchar(200),		-- 문의사항
    cost varchar(50)			-- 총비용
);

SELECT * FROM Reserve;
TRUNCATE Reserve;
DROP TABLE Reserve;
